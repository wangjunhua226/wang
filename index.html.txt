    <div id="info">📍 正在获取位置和方向...</div>
    <canvas id="compass" width="400" height="400"></canvas>
    <div id="hint">🛰️ 转动手机，让指针重合</div>
    <button class="permission" onclick="requestPermission()">🔐 重新授权传感器 (iOS需点此)</button>

    <script>
        // ========== 微信环境检测与引导 ==========
        (function() {
            if(navigator.userAgent.indexOf('MicroMessenger') !== -1) {
                document.getElementById('wechatTip').style.display = 'block';
            }
        })();

        // ========== 卫星参数（中星9号，东经 92.2°） ==========
        const SAT_LON = 92.2;
        let userLat = null, userLon = null;
        let satAzimuth = 0;
        let satElevation = 0;
        let deviceAlpha = 0;
        let permissionGranted = false;

        const canvas = document.getElementById('compass');
        const ctx = canvas.getContext('2d');
        const infoDiv = document.getElementById('info');
        const hintDiv = document.getElementById('hint');

        // ========== 方位角与仰角计算 ==========
        function calcAngles(lat, lon, satLon) {
            const toRad = Math.PI / 180;
            const toDeg = 180 / Math.PI;
            const latRad = lat * toRad;
            const deltaLon = (lon - satLon) * toRad;
            
            let az = Math.atan2(Math.sin(deltaLon), 
                     Math.cos(latRad) * Math.tan(0) - Math.sin(latRad) * Math.cos(deltaLon)) * toDeg;
            if (az < 0) az += 360;

            const r = 6371;
            const R = 42164;
            const cosLat = Math.cos(latRad);
            const cosDelta = Math.cos(deltaLon);
            const distance = Math.sqrt(r*r + R*R - 2*r*R*cosLat*cosDelta);
            const el = Math.asin((R*R - r*r - distance*distance) / (2*r*distance)) * toDeg;
            return { azimuth: az, elevation: el };
        }

        // ========== 获取位置 ==========
        function getLocation() {
            if (!navigator.geolocation) {
                infoDiv.textContent = '❌ 浏览器不支持GPS，请更换浏览器';
                return;
            }
            navigator.geolocation.getCurrentPosition(pos => {
                userLat = pos.coords.latitude;
                userLon = pos.coords.longitude;
                updateSatData();
                infoDiv.textContent = `📍 纬度:${userLat.toFixed(4)}  经度:${userLon.toFixed(4)}`;
            }, err => {
                infoDiv.textContent = '⚠️ 无法获取位置，请允许位置权限或更换浏览器';
                userLat = 33.7521;
                userLon = 114.8677;
                updateSatData();
                infoDiv.textContent = `📍 演示坐标 纬度:${userLat} 经度:${userLon} (仅供测试)`;
            }, { enableHighAccuracy: true });
        }

        function updateSatData() {
            if (userLat == null || userLon == null) return;
            const res = calcAngles(userLat, userLon, SAT_LON);
            satAzimuth = res.azimuth;
            satElevation = res.elevation;
            hintDiv.textContent = `🛰️ 中星9号 方位:${satAzimuth.toFixed(1)}°  仰角:${satElevation.toFixed(1)}°`;
        }

        // ========== 陀螺仪方向 ==========
        function handleOrientation(event) {
            if (!permissionGranted) return;
            let alpha = event.alpha;
            if (alpha == null) return;
            if (event.webkitCompassHeading !== undefined) {
                alpha = event.webkitCompassHeading;
            }
            deviceAlpha = alpha % 360;
            drawCompass();
        }

        function requestPermission() {
            if (typeof DeviceOrientationEvent !== 'undefined' && 
                typeof DeviceOrientationEvent.requestPermission === 'function') {
                DeviceOrientationEvent.requestPermission()
                    .then(state => {
                        if (state === 'granted') {
                            permissionGranted = true;
                            window.addEventListener('deviceorientation', handleOrientation);
                            infoDiv.textContent = '✅ 传感器已授权，请转动手机';
                        } else {
                            infoDiv.textContent = '❌ 传感器权限被拒绝';
                        }
                    }).catch(() => {
                        infoDiv.textContent = '⚠️ 权限请求错误，请用浏览器打开';
                    });
            } else {
                permissionGranted = true;
                window.addEventListener('deviceorientation', handleOrientation);
                infoDiv.textContent = '✅ 传感器已启动 (安卓直接可用)';
            }
        }

        // ========== 绘制罗盘 ==========
        function drawCompass() {
            const w = canvas.width;
            const h = canvas.height;
            const centerX = w / 2;
            const centerY = h / 2;
            const radius = Math.min(w, h) * 0.42;

            ctx.clearRect(0, 0, w, h);

            // 背景盘
            ctx.beginPath();
            ctx.arc(centerX, centerY, radius + 8, 0, 2 * Math.PI);
            ctx.fillStyle = '#16213e';
            ctx.fill();
            ctx.strokeStyle = '#e0a800';
            ctx.lineWidth = 3;
            ctx.stroke();

            // 外圈刻度
            ctx.save();
            ctx.translate(centerX, centerY);
            for (let i = 0; i < 360; i += 5) {
                const rad = i * Math.PI / 180;
                const x1 = (radius - 5) * Math.sin(rad);
                const y1 = -(radius - 5) * Math.cos(rad);
                const x2 = (radius - 20) * Math.sin(rad);
                const y2 = -(radius - 20) * Math.cos(rad);
                ctx.beginPath();
                ctx.moveTo(x1, y1);
                ctx.lineTo(x2, y2);
                ctx.strokeStyle = i % 15 === 0 ? '#f0c040' : '#555';
                ctx.lineWidth = i % 15 === 0 ? 2 : 1;
                ctx.stroke();
            }
            const dirs = [
                { angle: 0, label: '北' },
                { angle: 90, label: '东' },
                { angle: 180, label: '南' },
                { angle: 270, label: '西' }
            ];
            ctx.font = 'bold 14px system-ui';
            ctx.fillStyle = '#fff';
            ctx.textAlign = 'center';
            ctx.textBaseline = 'middle';
            dirs.forEach(d => {
                const rad = d.angle * Math.PI / 180;
                const x = (radius - 28) * Math.sin(rad);
                const y = -(radius - 28) * Math.cos(rad);
                ctx.fillText(d.label, x, y);
            });
            ctx.restore();

            // 手机指向 (红色箭头)
            ctx.save();
            ctx.translate(centerX, centerY);
            const phoneRad = deviceAlpha * Math.PI / 180;
            ctx.rotate(phoneRad);
            ctx.beginPath();
            ctx.moveTo(0, -radius + 5);
            ctx.lineTo(-10, -radius - 15);
            ctx.lineTo(10, -radius - 15);
            ctx.closePath();
            ctx.fillStyle = '#e74c3c';
            ctx.fill();
            ctx.restore();

            // 卫星方位 (金色箭头)
            ctx.save();
            ctx.translate(centerX, centerY);
            const satRad = satAzimuth * Math.PI / 180;
            ctx.rotate(satRad);
            ctx.beginPath();
            ctx.moveTo(0, -radius + 10);
            ctx.lineTo(-6, -radius - 25);
            ctx.lineTo(6, -radius - 25);
            ctx.closePath();
            ctx.fillStyle = '#f0c040';
            ctx.fill();
            ctx.beginPath();
            ctx.moveTo(0, 0);
            ctx.lineTo(0, -radius + 15);
            ctx.strokeStyle = '#f0c040';
            ctx.lineWidth = 2;
            ctx.stroke();
            ctx.restore();

            // 中心点
            ctx.beginPath();
            ctx.arc(centerX, centerY, 8, 0, 2*Math.PI);
            ctx.fillStyle = '#fff';
            ctx.fill();

            // 对准检测
            let diff = Math.abs(deviceAlpha - satAzimuth);
            if (diff > 180) diff = 360 - diff;
            ctx.font = '16px system-ui';
            ctx.fillStyle = '#fff';
            ctx.fillText(`${diff.toFixed(1)}°`, centerX, centerY + 30);

            if (diff < 2.5) {
                document.body.style.backgroundColor = '#1b3b1b';
                try {
                    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                    const osc = audioCtx.createOscillator();
                    const gain = audioCtx.createGain();
                    osc.connect(gain);
                    gain.connect(audioCtx.destination);
                    osc.frequency.value = 800;
                    gain.gain.value = 0.1;
                    osc.start();
                    osc.stop(audioCtx.currentTime + 0.12);
                } catch(e) {}
            } else {
                document.body.style.backgroundColor = '#1a1a2e';
            }
        }

        // ========== 启动 ==========
        function init() {
            getLocation();
            // 默认也请求一次权限（安卓会直接启用）
            requestPermission();
            // 持续绘制
            setInterval(() => {
                if (userLat != null && userLon != null) {
                    drawCompass();
                }
            }, 200);
        }

        init();
    </script>
